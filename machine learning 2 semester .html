# Atributos derivados teis
df['year'] = df['date'].dt.year
df['month'] = df['date'].dt.month

# idade aproximada no momento da venda
df['house_age'] = df['year'] - df['yr_built']

# preo por p quadrado
df['price_per_sqft'] = df['price'] / df['sqft_living']

# remover registros com valores no realistas ou nulos crticos
cols_core = ['price','sqft_living','lat','long','zipcode','year','month','house_age','price_per_sqft']
df_clean = df.dropna(subset=cols_core).copy()

# cortar outliers extremos de price_per_sqft (1%99%) para clustering mais estvel
low, high = df_clean['price_per_sqft'].quantile([0.01, 0.99])
df_clean = df_clean[(df_clean['price_per_sqft']>=low) & (df_clean['price_per_sqft']<=high)]

# Avaliar KMeans para K=2..10
results = []
for k in range(2, 11):
    km = KMeans(n_clusters=k, n_init=10, random_state=42)
    labels = km.fit_predict(X_scaled)
    sil = silhouette_score(X_scaled, labels)
    dbi = davies_bouldin_score(X_scaled, labels)
    results.append((k, sil, dbi))

res_df = pd.DataFrame(results, columns=['k','silhouette','davies_bouldin']).sort_values('k')

import itertools
import statsmodels.api as sm

def fit_aic(train_series, pdq):
    try:
        model = sm.tsa.ARIMA(train_series, order=pdq)
        res = model.fit()
        return res.aic, res
    except:
        return np.inf, None

pdq_grid = list(itertools.product(range(0,4), range(0,3), range(0,4)))
scores = []
for pdq in pdq_grid:
    aic, res = fit_aic(train, pdq)
    scores.append((pdq, aic))

scores_sorted = sorted(scores, key=lambda x: x[1])
